Phase 2f / Phase 2d, the IF infrastructure from Phase 6.1, the MIDI plumbing from Phase 6.3.

```rust
fn process_fm_network(
    &mut self,
    bins: &mut [Complex<f32>],
    curves: &[&[f32]],
    ctx: &ModuleContext,
) {
    use crate::dsp::modules::harmony_helpers::{find_top_k_peaks, PeakRecord};

    let n = bins.len();
    let amount = curves.get(0).copied().unwrap_or(&[]);
    let amt_centre = amount.get(n / 2).copied().unwrap_or(0.0).clamp(0.0, 1.0);
    if amt_centre < 1e-9 { return; }

    let if_buf = match ctx.instantaneous_freq {
        Some(b) if b.len() >= n => b,
        _ => return, // Passthrough if IF infra not active.
    };

    let threshold = curves.get(1).copied().unwrap_or(&[]);
    let thr_centre = threshold.get(n / 2).copied().unwrap_or(0.1);

    // Pre-fill magnitude scratch.
    self.scratch_mag.resize(n, 0.0);
    for k in 0..n { self.scratch_mag[k] = bins[k].norm(); }

    // Find up to 16 loudest peaks above threshold.
    self.fm_partials.fill(PeakRecord::default());
    let n_partials = find_top_k_peaks(
        &self.scratch_mag[..n],
        thr_centre,
        &mut self.fm_partials,
    );
    self.fm_partial_count = n_partials;

    // Cache per-partial frequency from IF buffer.
    for p in 0..n_partials {
        let bin = self.fm_partials[p].bin;
        self.fm_partial_freq[p] = if_buf.get(bin).copied().unwrap_or(0.0);
    }

    // Task 3 wires the modulation pass below.
    let _ = ctx;
}
```

## Task 3: FM Network — partial-pair frequency modulation

**Files:**
- Modify: `src/dsp/modules/modulate.rs` (`process_fm_network`)

For each detected partial, pick a "modulator" partial (the next one in the loudness list,
wrapping). Frequency-modulate the carrier's per-bin amplitude by the modulator's amplitude
× COEFFICIENT (modulation depth). The result: a 16-op-DX-style metallic re-spectralisation
of the input.

Implementation (per partial pair):
```
modulated_mag = carrier_mag × (1 + COEFFICIENT × modulator_mag × cos(modulator_phase))
```

Cosine of the modulator phase produces sidebands when convolved with the carrier in
log-frequency space. We emit the modulated magnitude at the carrier's bin, plus two
side-band bins at `carrier_freq ± modulator_freq`.

- [ ] **Step 3: Implement modulation pass**

Replace the trailing `let _ = ctx;` in `process_fm_network` with:
```rust
let coefficient = curves.get(4).copied().unwrap_or(&[]);
let mix         = curves.get(5).copied().unwrap_or(&[]);
let bin_freq    = ctx.sample_rate / ctx.fft_size as f32;

if self.fm_partial_count < 2 { return; }

// For each carrier, pick (i + 1) mod count as modulator.
for i in 0..self.fm_partial_count {
    let j = (i + 1) % self.fm_partial_count;
    let carrier   = self.fm_partials[i];
    let modulator = self.fm_partials[j];
    if carrier.bin == modulator.bin { continue; }

    let depth = coefficient.get(carrier.bin).copied().unwrap_or(0.0).clamp(0.0, 2.0) * 0.5;
    let mix_v = mix.get(carrier.bin).copied().unwrap_or(1.0).clamp(0.0, 1.0);

    let mod_freq    = self.fm_partial_freq[j];
    let carrier_freq = self.fm_partial_freq[i];
    if carrier_freq <= 0.0 || mod_freq <= 0.0 { continue; }

    let carrier_phase = bins[carrier.bin].arg();
    let mod_phase     = bins[modulator.bin].arg();

    // Modulate the carrier amplitude in place.
    let am_factor = 1.0 + depth * modulator.mag * mod_phase.cos();
    let new_carrier_mag = carrier.mag * am_factor.max(0.0);
    let new_carrier = Complex::from_polar(new_carrier_mag, carrier_phase);
    bins[carrier.bin] = bins[carrier.bin] * (1.0 - mix_v)
                      + new_carrier * mix_v;

    // Emit sidebands at carrier ± modulator frequencies.
    let upper_freq = carrier_freq + mod_freq;
    let lower_freq = (carrier_freq - mod_freq).abs();
    let upper_bin = (upper_freq / bin_freq + 0.5) as usize;
    let lower_bin = (lower_freq / bin_freq + 0.5) as usize;
    let sb_mag = depth * carrier.mag * modulator.mag * 0.5 * mix_v;
    if upper_bin > 0 && upper_bin < bins.len() {
        bins[upper_bin] += Complex::from_polar(sb_mag, carrier_phase);
    }
    if lower_bin > 0 && lower_bin < bins.len() && lower_bin != carrier.bin {
        bins[lower_bin] += Complex::from_polar(sb_mag, carrier_phase);
    }
}
```

## Task 6: NoteIn branch in Arpeggiator process arm

**Files:**
- Modify: `src/dsp/modules/rhythm.rs` (Arpeggiator arm)

When `arp_trigger_source == NoteIn`, the step advance is driven by MIDI NoteOn events

    let advance = match self.arp_trigger_source {
        ArpTriggerSource::Bpm => {
            // Existing BPM-driven advance, untouched.
            self.bpm_advance_count(ctx)
        }
        ArpTriggerSource::NoteIn => {
            // Count rising edges in the held-notes bitmap since last block.
            let mut new_count = 0_usize;
            if let Some(held) = ctx.midi_held_notes {
                for k in 0..128 {
                    if held[k] && !self.arp_prev_held_notes[k] {
                        new_count += 1;
                    }
                }
                self.arp_prev_held_notes.copy_from_slice(held);
            }
            new_count
        }
    };

    self.arp_step = (self.arp_step + advance) % crate::dsp::modules::rhythm::ARP_STEPS;

    // Existing step→envelope kernel runs as before, indexed by self.arp_step.
    self.apply_arp_step_envelope(bins, curves, ctx);
}
